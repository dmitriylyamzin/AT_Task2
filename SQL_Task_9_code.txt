SELECT ST_ABONENTS.C_NAME AS ABONENTNAME, EXTRACT(HOUR FROM TO_TIMESTAMP(QRY_QUEUE.C_EXEC_TIME, 'DD-MM-YYYY HH24:MI:SS' )) AS HH24, 
    TO_CHAR(ROUND (COUNT(*)/360, 2), '90.99') AS CNT FROM QRY_QUEUE 
    JOIN QRY_TYPE ON QRY_QUEUE.C_QRY_TYPE = QRY_TYPE.ID  
    JOIN ST_ABONENTS ON ST_ABONENTS.ID = QRY_TYPE.C_AB_REF
    WHERE QRY_QUEUE.C_ST = 1 OR QRY_QUEUE.C_ST = 0
    GROUP BY ST_ABONENTS.C_NAME, EXTRACT(HOUR FROM TO_TIMESTAMP(QRY_QUEUE.C_EXEC_TIME, 'DD-MM-YYYY HH24:MI:SS' ))
    ORDER BY ST_ABONENTS.C_NAME;
