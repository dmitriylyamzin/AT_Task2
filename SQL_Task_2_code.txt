SELECT ST_ABONENTS.C_NAME AS ABONENTNAME, COUNT(*) AS CNT FROM QRY_QUEUE 
    JOIN QRY_TYPE ON QRY_QUEUE.C_QRY_TYPE = QRY_TYPE.ID 
    JOIN ST_ABONENTS ON ST_ABONENTS.ID = QRY_TYPE.C_AB_REF 
    WHERE QRY_QUEUE.C_IN_TIME >= '01-01-2015 09:00:00' AND QRY_QUEUE.C_IN_TIME <= '01-01-2015 18:00:00' 
    GROUP BY ST_ABONENTS.C_NAME;
